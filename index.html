async function executarIAGlobal() { let apiKey = localStorage.getItem('gemini-api-key'); if (!apiKey) { alert("Configure a chave de API na aba de Estudos primeiro!"); return; } let promptText = document.getElementById('global-ai-prompt').value; if (!promptText.trim()) return; let btn = document.getElementById('btn-global-ai'); let status = document.getElementById('global-ai-status'); btn.disabled = true; btn.style.opacity = '0.6'; status.style.display = 'block'; status.style.color = 'var(--bordo-personalidade)'; status.innerText = 'Processando seu comando...'; let systemInstruction = `Você é o assistente virtual do planner. O usuário enviará comandos para ADICIONAR ou APAGAR itens. Retorne APENAS um objeto JSON válido, sem markdown. Estrutura: { "actions": [ { "type": "todo", "dayOffset": 0, "text": "Nome", "action": "add" }, { "type": "wishlist", "category": "quero", "text": "Nome", "action": "add" }, { "type": "calendar", "year": 2026, "month": 6, "day": 15, "text": "Evento", "color": "#cfe7e5", "action": "add" }, { "type": "clear_schedule", "action": "delete" } ] } Regras: - Se o usuário pedir para apagar matérias ou cronograma, use {"type": "clear_schedule", "action": "delete"}. - todo dayOffset: 0(Segunda) a 5(Sábado). - Se não houver itens, omita a ação. Retorne apenas o JSON. Hoje é: ` + new Date().toLocaleDateString('pt-BR'); try { let response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${apiKey}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: systemInstruction + "\nTexto: " + promptText }] }], generationConfig: { responseMimeType: "application/json" } }) }); let data = await response.json(); if (data.error) throw new Error(data.error.message); let resultJSON = JSON.parse(data.candidates[0].content.parts[0].text); if (resultJSON.actions) { resultJSON.actions.forEach(acao => { if (acao.action === 'delete' && acao.type === 'clear_schedule') { localStorage.removeItem('study-data'); } else if (acao.action === 'add') { if (acao.type === 'todo') { let tasks = JSON.parse(localStorage.getItem('todo-tasks')) || []; let d = new Date(getMondayDate(realToday)); d.setDate(d.getDate() + acao.dayOffset); tasks.push({ text: acao.text, done: false, date: d.toISOString().split('T')[0], notes: '' }); localStorage.setItem('todo-tasks', JSON.stringify(tasks)); } else if (acao.type === 'wishlist') { let cat = acao.category || 'quero'; let wishes = JSON.parse(localStorage.getItem(`wish-${cat}`)) || []; wishes.push({ text: acao.text, done: false }); localStorage.setItem(`wish-${cat}`, JSON.stringify(wishes)); } else if (acao.type === 'calendar') { let key = `cal-${acao.year}-${acao.month}-${acao.day}`; let events = JSON.parse(localStorage.getItem(key) || '[]'); events.push({ text: acao.text, color: acao.color || '#cfe7e5' }); localStorage.setItem(key, JSON.stringify(events)); } } }); atualizarTelas(); renderTodoList(); renderWishlist(); loadStudies(); status.innerText = 'Magia concluída!'; status.style.color = '#2e7d32'; document.getElementById('global-ai-prompt').value = ''; } } catch (err) { console.error(err); status.style.color = '#c62828'; status.innerText = 'Erro ao processar.'; } finally { btn.disabled = false; btn.style.opacity = '1'; setTimeout(() => { status.style.display = 'none'; }, 2000); } }
